<script>
    (function() {
        const ADMIN_PASS = '2026';
        const avatarImg = document.getElementById('avatarImage');
        const placeholder = document.getElementById('avatarPlaceholder');
        const fileInput = document.getElementById('avatarUpload');
        const unlockBtn = document.getElementById('unlockUploadBtn');
        const passInput = document.getElementById('adminPass');
        const adminPanel = document.getElementById('adminUploadPanel');
        const showAdminBtn = document.getElementById('showAdminBtn');
        const avatarWrapper = document.getElementById('avatarWrapper');

        // Toggle upload panel
        showAdminBtn.addEventListener('click', () => {
            adminPanel.style.display = adminPanel.style.display === 'block' ? 'none' : 'block';
        });

        // Verify password and trigger file selection
        unlockBtn.addEventListener('click', () => {
            if (passInput.value === ADMIN_PASS) {
                fileInput.disabled = false;
                fileInput.click();
            } else {
                alert('Incorrect password. Access denied.');
            }
        });

        // Handle file selection
        fileInput.addEventListener('change', (e) => {
            const file = fileInput.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = (event) => {
                    avatarImg.src = event.target.result;
                    avatarImg.style.display = 'block';
                    placeholder.style.display = 'none';
                    localStorage.setItem('yuhong_avatar', event.target.result);
                };
                reader.readAsDataURL(file);
            }
            passInput.value = '';
            fileInput.disabled = true;
            adminPanel.style.display = 'none';
        });

        // Click on avatar wrapper toggles panel (convenience)
        avatarWrapper.addEventListener('click', (e) => {
            if (e.target.closest('button') || e.target.closest('input')) return;
            adminPanel.style.display = adminPanel.style.display === 'block' ? 'none' : 'block';
        });

        // Load saved avatar
        const savedAvatar = localStorage.getItem('yuhong_avatar');
        if (savedAvatar) {
            avatarImg.src = savedAvatar;
            avatarImg.style.display = 'block';
            placeholder.style.display = 'none';
        }

        // Enter key in password field triggers verification
        passInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                unlockBtn.click();
            }
        });
    })();
</script>
